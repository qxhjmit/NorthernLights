begin
 for j in (SELECT max(hd.pfh_id) as pfh_id,
               hd.pfh_mer_no,
               hd.pfh_gw_no,
               max(hd.pfh_version)
          FROM ccps_prc_fee_hd hd
         where hd.pfh_status = 4
         group by hd.pfh_mer_no, hd.pfh_gw_no)     
         loop
   delete from ccps_prc_fee_tier t
    where exists (select h.param_id
          from ccps_prc_feelist_param h
         where h.pfh_id = j.pfh_id);
   delete from CCPS_prc_FEELIST_PARAM fp where fp.pfh_id = j.pfh_id;
   for i in (select * from ccps_feelist_param p where p.gw_no = j.pfh_gw_no) loop
   insert into ccps_prc_feelist_param
    (param_id,
     PFH_ID,
     feelist_id,
     contient,
     country,
     cardtyp,
     chrgmth,
     chrgcle,

    (sq_ccps_feelist_param_seq.nextval,
     j.pfh_id,
     i.feelist_id,
     i.contient,
     i.country,
     i.cardtyp,
     i.chrgmth,
     i.chrgcle,
     i.stardate,
     i.tier_flg,
     i.chrgpct,
     i.chrgval,
     i.aft_chrgpct,
     i.aft_chrgval,
     i.tier_mth,

     i.succ_flag,
     i.minchrgval,
     i.chrg_apply_flag,
     i.maxchrgval,
     i.chrg_currency,
     i.chrgval_currency);
  for k in (select * from ccps_fee_tier t where t.param_id = i.param_id) loop
    insert into ccps_prc_fee_tier
      (tier_id,
       param_id,
       tier_frm,
       tier_to,
       tier_pct,
       tier_val,

       k.tier_to,
       k.tier_pct,
       k.tier_val,
       k.tier_startdate,
       k.tier_aft_pct,
       k.tier_aft_val);
  end loop;
end loop;
end loop;
end ;

java你不知道的list转换!

发表于 2019-05-25

时间:2019-04-24

String[] –> List转化形式:

String [] arr = new String[] {“1”, “2”};

List list = Arrays.asList(arr);

由于String[]转化为list的时候存储空间已经固定为2,所以不能是用list中的list.remove(i)方法,如果想要
使用remove(i)这个方式,必须开辟一个新的空间.

如何开辟新的空间?

形式如下:

List list = new ArrayList(Arrays.asList(arr));

    var bottHtml = "</body>";        
    $(".logo-img").show();
    var trLength = $('.content-download table tr').length;
    var heightContent = $('.content-download').height();
    var heightDifference = heightContent - (trLength + 1) * 32;
    if (heightDifference > 0) {
        var $tdPadding = $('.content-download table tr td');
        var tdLength = $tdPadding.length - 1;
        var padding = $tdPadding.eq(tdLength).css('padding').substring(0,
                $tdPadding.eq(tdLength).css('padding').length - 2);
        var paddingValue = padding - heightDifference / trLength > 0.5 ? padding
                - heightDifference / trLength
                : 0.5;
        $tdPadding.css('padding', paddingValue);
    }

     bdhtml = window.document.body.innerHTML; //获取当前页的html代码
     var  sprnstr="<!--startprint-->";//设置打印开始区域
     var  eprnstr="<!--endprint-->";//设置打印结束区域
     prnhtml = headHtml + bdhtml.substring(bdhtml.indexOf(sprnstr) + 18); //从开始代码向后取html
     prnhtml = prnhtml.substring(0, prnhtml.indexOf(eprnstr)) + bottHtml; //从结束代码向前取html

     window.document.body.innerHTML = prnhtml;       
    pagesetup_null();

}

var hkey_root, hkey_path, hkey_key;

function pagesetup_null() {
    try {
        var RegWsh = new ActiveXObject("WScript.Shell");
        hkey_key = "header";
        RegWsh.RegWrite(hkey_root + hkey_path + hkey_key, "");
        hkey_key = "footer";
        RegWsh.RegWrite(hkey_root + hkey_path + hkey_key, "");
    } catch (e) {
    }
}
