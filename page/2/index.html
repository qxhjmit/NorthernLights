#创建表：
　CREATE TABLE Student(

　　　　id varchar2(32) primary key,

　　　　name varchar2(8) not null,

　　　　age number

　　);

#修改字段名:
alter table Student rename name to StuName;

#修改数据类型：
alter table Student modify (id varchar2(64));

#添加字段
alter table 表名 add 新字段名 类型+长度

例如:alter table test add pid varchar2(10);

#添加备注
comment on column test.pid IS ‘测试备注’;

#删除字段
ALTER TABLE ccps_report_hd DROP COLUMN RH_MENU_POSITION;

#修改增值序列
alter sequence sq_CCPS_TRADERECORD_STATUS increment by 1;

#字段名查表
select * from user_tab_columns t where t.COLUMN_NAME =’BCI_SEND_ORDERNO_FLAG’;

#分组
–使用分析函数row_number() over (partiion by … order by …)来进行分组编号，然后取分组标号值为1的记录即可。目前主流的数据库都有支持分析函数，很好用。

--其中，partition by 是指定按哪些字段进行分组，这些字段值相同的记录将在一起编号；order by则是指定在同一组中进行编号时是按照怎样的顺序。

select s.*  from ( 
    select pfh.*, row_number() over (partition by pfh.pfh_mer_no,pfh.pfh_gw_no,pfh.pfh_version order by pfh.pfh_version) as group_idx  
    from ccps_prc_fee_hd pfh where pfh.pfh_mer_no = 21580
) s
where s.group_idx = 1

#去重
select distinct(a.rf_key_type_desc) from ccps_report_format a

#删除 或者更新失误返回数据(不适用有外键的表)
select * from ccps_report_format as of timestamp to_timestamp(‘2018-05-11 16:48:11’,’yyyy-mm-dd hh24:mi:ss’)
alter table ccps_report_format enable row movement;
flashback table ccps_report_format to timestamp to_timestamp(‘2018-05-11 16:48:11’,’yyyy-mm-dd hh24:mi:ss’)

#合并多列成一个字符串，多行成一个字符串
select deptno,listagg(ename,’,’)within group(order by sal)name from emp group by deptno;

#过滤重复数据
select from xi a where (a.username) in (select username from xi group by username having count() > 1)

private static List<File> getFiles(String path) {
    File root = new File(path);
    List<File> files = new ArrayList<File>();
    if (!root.isDirectory()) {
        files.add(root);
    } else {
        File[] subFiles = root.listFiles();
        for (File f : subFiles) {
            files.addAll(getFiles(f.getAbsolutePath()));
        }
    }
    return files;
}

//通过冲入的list<object>创建excel表格.2017
private static String createExcel(List<? extends Object> list, String path) {
    String result = "";
    if (list.size() == 0 || list == null) {
        result = "没有对象信息";
    } else {
        Object o = list.get(0);
        Class<? extends Object> clazz = o.getClass();
        String className = clazz.getSimpleName();
        java.lang.reflect.Field[] fields = clazz.getDeclaredFields(); // 这里通过反射获取字段数组
        File folder = new File(path);
        if (!folder.exists()) {
            folder.mkdirs();
        }
        //获取String类型的时间
        String fileName = "";
        SimpleDateFormat df = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");//设置日期格式
        String dateStr = df.format(new Date());
        dateStr = dateStr.replaceAll("[-/:\\s]+", "");
        fileName = "exportI18n" + dateStr;
        String name = fileName.concat(".xls");
        WritableWorkbook book = null;
        File file = null;
        try {
            file = new File(path.concat(File.separator).concat(name));
            book = Workbook.createWorkbook(file); // 创建xls文件
            WritableSheet sheet = book.createSheet(className, 0);
            int i = 0; // 列
            int j = 0; // 行
            for (java.lang.reflect.Field f : fields) {
                j = 0;
                Label label = new Label(i, j, f.getName()); // 这里把字段名称写入excel第一行中
                sheet.addCell(label);
                j = 1;
                for (Object obj : list) {
                    Object temp = getFieldValueByName(f.getName(), obj);
                    String strTemp = "";
                    if (temp != null) {
                        strTemp = temp.toString();
                    }
                    sheet.addCell(new Label(i, j, strTemp)); // 把每个对象此字段的属性写入这一列excel中
                    j++;
                }
                i++;
            }
            book.write();
            result = file.getPath();
        } catch (Exception e) {
            result = "SystemException";
            e.printStackTrace();
        } finally {

            fileName = null;
            name = null;
            folder = null;
            file = null;

            if (book != null) {
                try {
                    book.close();
                } catch (WriteException e) {
                    result = "WriteException";
                    e.printStackTrace();
                } catch (IOException e) {
                    result = "IOException";
                    e.printStackTrace();
                }
            }
        }

    }

    return result; // 最后输出文件路径
}

//根据参数名生成栏位和数据
private static Object getFieldValueByName(String fieldName, Object o) {
    try {
        String firstLetter = fieldName.substring(0, 1).toUpperCase();
        String getter = "get" + firstLetter + fieldName.substring(1); // 获取方法名
        java.lang.reflect.Method method = o.getClass().getMethod(getter,
                new Class[]{}); // 获取方法对象
        Object value = method.invoke(o, new Object[]{}); // 用invoke调用此对象的get字段方法
        return value; // 返回值
    } catch (Exception e) {
        e.printStackTrace();
        return null;
    }
}

使用打印功能重新渲染页面,导致jq按钮功能失效!

05-09 用jq绑定的节点按钮失效.

var OldDdhtml = window.document.body.innerHTML;//获取当前页的html代码    
var prnhtml = "html";//这里随便定义,可以定义任意html内容
window.document.body.innerHTML = prnhtml; 
window.document.body.innerHTML = OldDdhtml;
使用上述方法改变页面内容,导致按钮点击事件失效.

原因分析:由于直接使用节点绑定事件,当页面重新内容替换,虽然没有改变布局,但是节点的点击事件并没有重新绑定,导致了按钮点击事件失效.

解决方法:

第一种方法:重新刷新当前页 —– window.location.reload();

第二种方法:如果你是用节点绑定事件,例如:$(‘.button’).clcik(function(){}),改用用document绑定$(document).on(‘click’,’.button’,function(){})这样就可以解决按钮事件失效.
为什么这样可以处理这个问题,这就是你直接操作节点,但是你页面并没有重新渲染,导致事件没有绑定,但是如果直接把时间绑定在document上,在去寻找这个节点,就可以找寻得到.

java你不知道的list转换!

String[] –> List转化形式:

String [] arr = new String[] {“1”, “2”};

List list = Arrays.asList(arr);

由于String[]转化为list的时候存储空间已经固定为2,所以不能是用list中的list.remove(i)方法,如果想要
使用remove(i)这个方式,必须开辟一个新的空间.

如何开辟新的空间?

形式如下:

List list = new ArrayList(Arrays.asList(arr));
