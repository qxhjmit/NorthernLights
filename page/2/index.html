java你不知道的秘密!

String[] –> List转化形式:

String [] arr = new String[] {“1”, “2”};

List list = Arrays.asList(arr);

由于String[]转化为list的时候存储空间已经固定为2,所以不能是用list中的list.remove(i)方法,如果想要
使用remove(i)这个方式,必须开辟一个新的空间.

如何开辟新的空间?

形式如下:

List list = new ArrayList(Arrays.asList(arr));

window.print()打印

打印固定区域功能实现

function dayin() {
    var OldDdhtml = window.document.body.innerHTML;//获取当前页的html代码,用作还原    
    //去掉页脚页眉
    var headHtml = "<head><style type='text/css' media='print'> @page{ size:  auto;   /* auto is the initial value */margin: 0mm;  /* this affects the margin in the printer settings */}html{/*background-color: #FFFFFF; */margin: 0px;  /* this affects the margin on the html before sending to printer */}</style></head><body>"
    var bottHtml = "</body>";        
    $(".logo-img").show();
    var trLength = $('.content-download table tr').length;
    var heightContent = $('.content-download').height();
    var heightDifference = heightContent - (trLength + 1) * 32;
    if (heightDifference > 0) {
        var $tdPadding = $('.content-download table tr td');
        var tdLength = $tdPadding.length - 1;
        var padding = $tdPadding.eq(tdLength).css('padding').substring(0,
                $tdPadding.eq(tdLength).css('padding').length - 2);
        var paddingValue = padding - heightDifference / trLength > 0.5 ? padding
                - heightDifference / trLength
                : 0.5;
        $tdPadding.css('padding', paddingValue);
    }

     bdhtml = window.document.body.innerHTML; //获取当前页的html代码
     var  sprnstr="<!--startprint-->";//设置打印开始区域
     var  eprnstr="<!--endprint-->";//设置打印结束区域
     prnhtml = headHtml + bdhtml.substring(bdhtml.indexOf(sprnstr) + 18); //从开始代码向后取html
     prnhtml = prnhtml.substring(0, prnhtml.indexOf(eprnstr)) + bottHtml; //从结束代码向前取html
     //var styleHtml =  " style = '' ";
     window.document.body.innerHTML = prnhtml;       
    pagesetup_null();
    window.print();
    window.document.body.innerHTML = OldDdhtml;
}

var hkey_root, hkey_path, hkey_key;
hkey_root = "HKEY_CURRENT_USER";
hkey_path = "\Software\Microsoft\Internet Explorer\PageSetup\ ";

function pagesetup_null() {
    try {
        var RegWsh = new ActiveXObject("WScript.Shell");
        hkey_key = "header";
        RegWsh.RegWrite(hkey_root + hkey_path + hkey_key, "");
        hkey_key = "footer";
        RegWsh.RegWrite(hkey_root + hkey_path + hkey_key, "");
    } catch (e) {
    }
}
