for循环

begin
 for j in (SELECT max(hd.pfh_id) as pfh_id,
               hd.pfh_mer_no,
               hd.pfh_gw_no,
               max(hd.pfh_version)
          FROM ccps_prc_fee_hd hd
         where hd.pfh_status = 4
         group by hd.pfh_mer_no, hd.pfh_gw_no)     
         loop
   delete from ccps_prc_fee_tier t
    where exists (select h.param_id
          from ccps_prc_feelist_param h
         where h.pfh_id = j.pfh_id);
   delete from CCPS_prc_FEELIST_PARAM fp where fp.pfh_id = j.pfh_id;
   for i in (select * from ccps_feelist_param p where p.gw_no = j.pfh_gw_no) loop
   insert into ccps_prc_feelist_param
    (param_id,
     PFH_ID,
     feelist_id,
     contient,
     country,
     cardtyp,
     chrgmth,
     chrgcle,
     stardate,
     tier_flg,
     chrgpct,
     chrgval,
     aft_chrgpct,
     aft_chrgval,
     tier_mth,
     tier_unit,
     succ_flag,
     minchrgval,
     chrg_apply_flag,
     maxchrgval,
     chrg_currency,
     chrgval_currency)
  values
    (sq_ccps_feelist_param_seq.nextval,
     j.pfh_id,
     i.feelist_id,
     i.contient,
     i.country,
     i.cardtyp,
     i.chrgmth,
     i.chrgcle,
     i.stardate,
     i.tier_flg,
     i.chrgpct,
     i.chrgval,
     i.aft_chrgpct,
     i.aft_chrgval,
     i.tier_mth,
     i.tier_unit,
     i.succ_flag,
     i.minchrgval,
     i.chrg_apply_flag,
     i.maxchrgval,
     i.chrg_currency,
     i.chrgval_currency);
  for k in (select * from ccps_fee_tier t where t.param_id = i.param_id) loop
    insert into ccps_prc_fee_tier
      (tier_id,
       param_id,
       tier_frm,
       tier_to,
       tier_pct,
       tier_val,
       tier_startdate,
       tier_aft_pct,
       tier_aft_val)
    values
      (sq_ccps_fee_tier_seq.nextval,
       sq_ccps_feelist_param_seq.currval,//取的为前面的索引
       k.tier_frm,
       k.tier_to,
       k.tier_pct,
       k.tier_val,
       k.tier_startdate,
       k.tier_aft_pct,
       k.tier_aft_val);
  end loop;
end loop;
end loop;
end ;
